Database Setup
Run these in Supabase → SQL Editor. Run each block separately.
📋 1 — embed_servers table (if not done yet)
CREATE TABLE IF NOT EXISTS embed_servers (
  id          bigint generated by default as identity primary key,
  name        text not null,
  emoji       text default '🌐',
  embed_url   text not null,
  tv_url      text default '',
  id_type     text default 'imdb',
  quality     text default 'HD',
  sort_order  int default 1,
  enabled     boolean default true,
  created_at  timestamptz default now()
);
ALTER TABLE embed_servers ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read" ON embed_servers FOR SELECT USING (enabled = true);
CREATE POLICY "anon write" ON embed_servers FOR ALL USING (true) WITH CHECK (true);
📋 2 — Add new columns (if table already existed)
ALTER TABLE embed_servers ADD COLUMN IF NOT EXISTS tv_url text default '';
ALTER TABLE embed_servers ADD COLUMN IF NOT EXISTS id_type text default 'imdb';
📋 3 — traffic table (NEW — for analytics)
CREATE TABLE IF NOT EXISTS traffic (
  id      bigint generated by default as identity primary key,
  date    date not null unique,
  visitors jsonb default '[]',
  count   int default 0
);
ALTER TABLE traffic ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read traffic" ON traffic FOR SELECT USING (true);
CREATE POLICY "anon write traffic" ON traffic FOR ALL USING (true) WITH CHECK (true);
📋 4 — site_settings table (NEW — for SEO/favicon)
CREATE TABLE IF NOT EXISTS site_settings (
  id    bigint generated by default as identity primary key,
  key   text not null unique,
  value text
);
ALTER TABLE site_settings ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read settings" ON site_settings FOR SELECT USING (true);
CREATE POLICY "anon write settings" ON site_settings FOR ALL USING (true) WITH CHECK (true);
📋 5 — user_meta table (for subscription + avatar tracking)
CREATE TABLE IF NOT EXISTS user_meta (
  id           bigint generated by default as identity primary key,
  user_id      uuid not null unique,
  email        text default '',
  display_name text default '',
  subscription text default 'free',
  avatar_url   text default '',
  created_at   timestamptz default now()
);
ALTER TABLE user_meta ENABLE ROW LEVEL SECURITY;
CREATE POLICY "users read own" ON user_meta FOR SELECT USING (true);
CREATE POLICY "anon all" ON user_meta FOR ALL USING (true) WITH CHECK (true);

-- If table already exists, add missing columns:
ALTER TABLE user_meta ADD COLUMN IF NOT EXISTS avatar_url text default '';
ALTER TABLE user_meta ADD COLUMN IF NOT EXISTS email text default '';
ALTER TABLE user_meta ADD COLUMN IF NOT EXISTS display_name text default '';
📋 6 — Backfill emails for existing users
Run this once to copy emails and names from Supabase Auth into user_meta for all existing users.
UPDATE user_meta um
SET
  email = au.email,
  display_name = COALESCE(
    au.raw_user_meta_data->>'full_name',
    split_part(au.email, '@', 1)
  )
FROM auth.users au
WHERE um.user_id = au.id
  AND (um.email IS NULL OR um.email = '');
